public class FixedDiscountStrategy implements DiscountStrategy {

    @Override
    public int execute(int price) {
        return price - 1000;
    }
}

public class RateDiscountStrategy implements DiscountStrategy {

    @Override
    public int execute(int price) {
        return price - (int)(price * 0.1);
    }
}
